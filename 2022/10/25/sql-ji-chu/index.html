SQL简介

SQL (Structured Query Language:结构化查询语言) 是用于管理关系数据库管理系统（RDBMS）。 SQL 的范围包括数据插入、查询、更新和删除，数据库模式创建和修改，以及数据访问控制。

  • SQL 指结构化查询语言，全称是 Structured Query Language。
  • SQL 让您可以访问和处理数据库，包括数据插入、查询、更新和删除。
  • SQL 在1986年成为 ANSI（American National Standards Institute 美国国家标准化组织）的一项标准，在 1987 年成为国际标准化组织（ISO）标准。

语法

-- 数据库中的字符串用单引号
-- 查看数据库版本
select version();
-- 显示当前数据库
select database();
-- 描述表结构 desc
describe 表名;
-- 查询字段
select 字段名1,字段2,.. from 表名;
-- 查询所有字段，\* 号效率低，或者写所有字段名
select * from 表名;
-- 给查询列起别名
select 字段名 as 别名 from 表名;
as 关键字可以省略
select 字段名 别名 from 表名;
别名有空格 select 字段名 'as 别名' from 表名;
-- 字段可以参与数学运算
select 字段\*12 from 表名;
-- 条件查询
select 字段名 from 表名 where 表名;
/* 条件 ：
等于：= 	不等于：<>	
与：and	字段名 between 左小 and 右大
或：or 
查询空条件
select 字段名 from 表名 where comm is none (not none);  不能用 = none
*/
-- in 相当于多个or,表示某个范围，not in 不在某个范围
select 字段名 from 表名 where name in(100,200); 表示100或200，不是区间
-- 模糊查询关键字like   % 表示任意多个字符 ，_ 表示任意一个字符
delect name from name_table where name like 'y%'; -- 表示以y开头的所有名字
delect name from name_table where name like '_y%'; -- 表示第二个字母是y的所有名字
select name from t where name like '%\_%'; -- 表示名字里有下划线的
-- 排序 关键字 order by
select 字段名 from 表名 order by 排序字段 asc; -- asc 升序 dsc 降序
select 字段名 from 表名 order by sal asc,ename asc; -- 当sal相同时，按照ename升序排
select 字段名 from 表名 order by 2; 	-- 按照第二列排序，不建议使用


-- 单行处理函数（数据处理函数） 一个输入对应一个输出
select lower(name) from student; -- 小写   大写：upper()
select substr(name,1,n) from student;  -- 截取子串，索引从1开始
select concat(No,Name) from student; -- 字符串拼接
length() -- 字符串长度
trim()  -- 去字符串中的空格
round(float,i)  -- 四舍五入,对float保留i位小数，-1 保留到10位，1保留十分位
rand() -- 生成0-1中的随机数
ifnull(num,temp)  -- 空值处理函数。若num值为空，当作temp处理。在数学运算中，若有空值，则结果位null。
case 字段 when 值1 then .. when 值2 then .. else .. end  -- 只起显示作用，不对数据库做修改


-- 分组函数  输入多行，输出一行，会自动忽略null，不需要提前对null处理
-- 若无分组，则一张表为一组
count()  -- 计数  count(*) 统计结果包括null，参数为具体字段则不包括null
sum()  -- 求和
avg()	-- 平均值
max()
min()
-- 分组函数不能直接使用在where子句中
select name,grade from student where grade>min(grade);  -- 此行语句错误
/*	语句执行顺序
from -> where -> group by -> select -> order by
where 执行时还没有分组，不能执行分组函数
*/
-- 分组查询
select .. from .. group by ..
select subject,avg(grade) from student group by subject;
-- select后只能跟参加分组的字段和分组函数。
-- 不同字段联合分组
select .. from .. group by 字段一,字段二;

-- 使用having可以对分完组的数据再次过滤  效率低
select subject,avg(grade) from student group by subject having max(grade)>60;
select subject,avg(grade) from student where grade>60 group by subject;   -- 优先选用
-- 去重关键字  distinct,出现在两个字段前，联合去重。
select distinct name from student;
select name,distinct school from student;  -- 语法错误

-- 连接查询 多张表跨表查询 
/* 
SQL92、SQL99
内连接
	等值连接
	非等值连接
	自连接
外连接
	左外连接（左连接）
	右外连接（右连接）
全连接
*/
-- 多表查询不加条件会出现笛卡尔积现象，效率低
select name,className from student,class where student.classNo=class.no;
select s.name,c.className from student s,class c where s.classNo=c.no;
-- 内连接-等值连接	inner可省略 join
-- SQL92
-- select name,className from student,class where student.classNo=class.no;
-- SQL99
select 
	name,className 
from 
	student 
join 
	class 
on 
	student.classNo=class.no
where
	筛选条件;
-- 内连接-非等值连接
select 
	name,className 
from 
	student 
join 
	class 
on 
	student.classNo between 1 and 3   	-- 班级号在1-3之间的
where
	筛选条件;
-- 内连接-自连接  一张表看做两张表
select s.name,b.name from student s join student b on s.banzhangNo=b.No;
-- 外连接 以左/右表为主表，查询主表所有信息，匹配附表 right outer可省略 join
-- 外连接-右连接
select 
	name,className 
from 
	student 
right join 
	class 
on 
	student.classNo=class.no
where
	筛选条件;
-- 全连接 两张表都是主表
-- 子查询  嵌套的select语句
select ..(select) from .. (select) where ..(select);

-- union合并查询结果集
select name,job from emp where job="manager"
union
select name,job from emp where job="salesman";  -- union 效率更高
-- 要求结果列数相同，类型相同

-- limit 将查询结果集中的一部风取出，通常使用在分页查询当中
select name,grade from student order by grade desc limit 0,5;  -- 下标，长度
-- limit在order by排序结果之后执行

-- 表的创建
create table 表名(
    字段名 数据类型,
    字段名,数据类型 default "默认值"
);
-- 快速建表
create table student2 as select * from student;
-- 常见数据类型
/*
varchar(255)  可变长度字符串，动态分配空间
char(255) 定长字符串
int(11)  整型
bigint	 长整型
float	 单精度浮点数
double	 双精度浮点数
date	 短日期
datetime 长日期
clob	 字符大对象，最多存储4G字符串
blob	 二进制大对象，专门存储图片、声音、视频等
*/

-- 删除表
drop table student; -- 当表不存在时报错
drop table if exists student;  -- 当表存在时删除

-- 插入数据
insert into 表名(字段名1,字段名2...) values(值1,值2...);
-- 未指定字段值为null
-- 省略insert中的字段名，相当于字段名默认添加，须按顺序和类型
-- insert 插入日期
format(数字，'格式')
str_to_date(字符串，'日期格式') -- %i分 %s秒
insert into student(birth) values(str_to_date('1990-10-1','%Y-%M-%D'));
-- date_format 查询日期  可以自动类型转换
select date_format(birth,'%Y/%M/%D') as birth from student;
-- 插入多条数据
insert into student(name,age) values
(name1,18),
(name2,19);
-- 查询结果插入表中
insert into student2 select * from student;

-- datetime  %h:%i:%s
now()  -- 系统当前时间

-- 修改数据 update() 无条件修改全表
update student set name ='jack',birthday='2001-11-05' where id=2;

-- 删除数据 无条件删除全表 删除数据但数据位置不回收，可以rollback恢复，但速率慢
delete from student where id=3;
-- 快速删除表中数据
truncate table student;-- 物理删除，速度快，不能回滚

-- 创建表时添加约束
